start = rep(expression, newlines, 1, 0)

ws = /\s+/
trim = /\s*/
newlines = /\s*\n\s*/

id = /[a-zA-Z0-9_]+/

statement =
  assignment

assignment = "let" ws id trim "=" trim expression

expression =
      value

string = ('"' /[^"]*/ '"') | ("'" /[^']*/ "'")
bool = "true" | "false"
int = /-?[0-9]+/
float = /-?([0-9]*\.)?[0-9]+f?/

value = string | bool | int | float
